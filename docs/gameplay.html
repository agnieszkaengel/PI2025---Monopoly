 89    def handle_events(self, event, screen):
 90        """
 91        Funkcja rozpoznaje rodzaj eventu i wywołuje odpowiednią funkcję obsługującą ten rodzaj lub zamyka program w przypadku kliknięcia klawisza Q.
 92        :param event: zdarzenie przechwycone w pętli głównej
 93        :param screen: płaszczyzna ekranu
 94        :return: None
 95        """
 96        if event.type == pygame.QUIT or (event.type == pygame.KEYDOWN and event.key == pygame.K_q):
 97            self.running = False
 98        elif event.type == pygame.MOUSEBUTTONDOWN:
 99            self.handle_mouse_click(event)
100        elif event.type == pygame.KEYDOWN:
101            self.handle_text_input(event)

152    def check_inbox_click(self, event):
153        """
154        Funkcja sprawdza, które pole do wprowadzania danych zostało kliknięte i aktywuje je.

159            for i, inbox in enumerate(self.menu.nick_inboxes):
160                if inbox.collidepoint(event.pos):
161                    self.nick_inbox_active = i

164        else:
165            for i, inbox in enumerate(self.menu.personalize_inboxes):
166                if inbox.collidepoint(event.pos):
167                    self.choice_inbox_active = i

358        """
359        Funkcja odpowiada za pętlę główną gry.
360        :param screen: płaszczyzna ekranu
361        :return: None
362        """
363        while self.running:
364            screen.fill((200, 220, 200))
365            self.screen = screen
366
367            for event in pygame.event.get():
368                self.handle_events(event, screen)
369
370            match self.current_state:
371                case GameState.MAIN_MENU.value:
372                    self.menu.draw_menu(screen)
373                case GameState.NICKNAME_MENU.value | GameState.CUSTOMIZE_PLAYERS.value:
374                    self.menu.create_users_list(self.players_number)
375                    self.menu.draw_nick_menu(screen, self.players_number, self.current_state)
376                case GameState.GAMEPLAY.value:
377                    self.update_gameplay(screen)
378                case GameState.PERSONALIZE_MENU.value:
379                    self.menu.personalize_game_menu(screen)
380                case GameState.STOP.value:
381                    name = self.players_singleton.find_winner()
382                    self.board_service.board.end_screen(screen, name)
